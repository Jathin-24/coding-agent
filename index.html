# Importing necessary libraries import os from selenium import webdriver from selenium.webdriver.common.keys import Keys from selenium.webdriver.support.ui import WebDriverWait from selenium.webdriver.support import expected_conditions as EC from selenium.webdriver.firefox.options import Options # Setting up the browser driver chrome_options = Options() chrome_options.add_argument('--headless') chrome_options.add_argument('--no-sandbox') chrome_options.add_argument('--disable-dev-shm-usage') driver = webdriver.Firefox(executable_path='/usr/bin/firefox', options=chrome_options) # Opening the coffee shop landing page in the browser url = 'https://www.example.com/coffee-shop' driver.get(url) # Waiting for the page to load WebDriverWait(driver, 10).until(EC.visibility_of_all_elements_located((By.CSS_SELECTOR, 'body'))) # Finding the "Sign Up" button and clicking on it signup_button = driver.find_element(By.CSS_SELECTOR, 'div.container-fluid.text-center.mt-5') signup_button.click() # Waiting for the login page to load WebDriverWait(driver, 10).until(EC.visibility_of_all_elements_located((By.CSS_SELECTOR, 'body'))) # Finding the "Login" button and clicking on it login_button = driver.find_element(By.CSS_SELECTOR, 'div.container-fluid.text-center.mt-5') login_button.click() # Waiting for the login form to load WebDriverWait(driver, 10).until(EC.visibility_of_all_elements_located((By.CSS_SELECTOR, 'body'))) # Finding the "Sign In" button and clicking on it signin_button = driver.find_element(By.CSS_SELECTOR, 'div.container-fluid.text-center.mt-5') signin_button.click() # Waiting for the login form to load WebDriverWait(driver, 10).until(EC.visibility_of_all_elements_located((By.CSS_SELECTOR, 'body'))) # Finding the "Logout" button and clicking on it logout_button = driver.find_element(By.CSS_SELECTOR, 'div.container-fluid.text-center.mt-5') logout_button.click() # Closing the browser driver.quit()
